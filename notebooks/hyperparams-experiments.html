1.3 K-Fold Cross Validation

import numpy as np
# Define the number of folds (k)
k = 5

# Initialize lists to store predictions and accuracies
predictions = {}
accuracies = []

# Calculate the size of each fold
fold_size = len(X) // k

# Perform k-fold cross-validation
for i in range(k):
    # Split the data into training and test sets
    test_start = i * fold_size
    test_end = (i + 1) * fold_size
    test_set = X[test_start:test_end]
    test_labels = y[test_start:test_end]
    
    training_set = np.concatenate((X[:test_start], X[test_end:]), axis=0)
    training_labels = np.concatenate((y[:test_start], y[test_end:]), axis=0)
    
    # Train the model
    dt_classifier = DecisionTreeClassifier(random_state=42)
    dt_classifier.fit(training_set, training_labels)
    
    # Make predictions on the validation set
    fold_predictions = dt_classifier.predict(test_set)
    
    # Calculate the accuracy of the fold
    fold_accuracy = np.mean(fold_predictions == test_labels)
    
    # Store the predictions and accuracy of the fold
    predictions[i] = fold_predictions
    accuracies.append(fold_accuracy)

# Print the predictions and accuracies of each fold
for i in range(k):
    print("Fold {}: Accuracy: {:.4f}".format(i+1, accuracies[i]))

2.2 Nested Cross-Validation

# Define the number of outer and inner loops
num_outer_folds = 5
num_inner_folds = 5

# Initialize lists to store the optimal hyperparameters and test accuracies
optimal_hyperparameters = []
test_accuracies = []

# Perform nested cross-validation
for i in range(num_outer_folds):
    # Split the data into outer training and test sets
    X_train_outer, X_test_outer, y_train_outer, y_test_outer = train_test_split(X, y, test_size=0.3, random_state=i)
    
    # Initialize variables to store the best hyperparameters and average validation accuracy
    best_accuracy = 0
    best_hyperparameters = {}
    fold_size = len(X_train_outer) // num_inner_folds
    
    # Iterate over the hyperparameter values
    for max_depth in hyperparameters['max_depth']:
        for min_samples_split in hyperparameters['min_samples_split']:
            for criterion in hyperparameters['criteria_values']:

                # Initialize a list to store the validation accuracies
                validation_accuracies = []
                
                # Perform inner cross-validation
                for j in range(num_inner_folds):

                    # Split the data into training and test sets
                    val_start = j * fold_size
                    val_end = (j + 1) * fold_size
                    val_set = X_train_outer[val_start:val_end]
                    val_labels = y_train_outer[val_start:val_end]
                    
                    training_set = np.concatenate((X_train_outer[:val_start], X_train_outer[val_end:]), axis=0)
                    training_labels = np.concatenate((y_train_outer[:val_start], y_train_outer[val_end:]), axis=0)
                    
                    # Train the model
                    dt_classifier = DecisionTreeClassifier(max_depth=max_depth, min_samples_split=min_samples_split, criterion=criterion, random_state=42)
                    dt_classifier.fit(training_set, training_labels)
                    
                    # Make predictions on the validation set
                    val_accuracy = dt_classifier.score(val_set, val_labels)
                                        
                    # Append the validation accuracy to the list
                    validation_accuracies.append(val_accuracy)
                
                # Calculate the average validation accuracy
                avg_val_accuracy = np.mean(validation_accuracies)
                
                # Update the best accuracy and hyperparameters
                if avg_val_accuracy > best_accuracy:
                    best_accuracy = avg_val_accuracy
                    best_hyperparameters = {
                        'max_depth': max_depth,
                        'min_samples_split': min_samples_split,
                        'criterion': criterion
                    }
    
    # Store the optimal hyperparameters for the current outer loop
    optimal_hyperparameters.append(best_hyperparameters)
    
    # Train the model with the best hyperparameters on the outer training set
    dt_classifier = DecisionTreeClassifier(max_depth=best_hyperparameters['max_depth'], min_samples_split=best_hyperparameters['min_samples_split'], criterion=best_hyperparameters['criterion'], random_state=42)
    dt_classifier.fit(X_train_outer, y_train_outer)
    
    # Evaluate the performance on the outer test set
    test_accuracy = dt_classifier.score(X_test_outer, y_test_outer)
    test_accuracies.append(test_accuracy)

# Print the optimal hyperparameters and test accuracies for each outer loop
for i in range(num_outer_folds):
    print("Optimal hyperparameters for outer loop {}: {}".format(i+1, optimal_hyperparameters[i]))
    print("Test accuracy for outer loop {}: {:.4f}".format(i+1, test_accuracies[i]))
